06 — 手写 call / apply / bind / new

这四个是面试最高频的手写题。核心思路：把函数临时挂到目标对象上，通过对象调用，this 就自动指向目标了。

一、手写 call

源码

// myCall(thisArg, arg1, arg2, ...)
Function.prototype.myCall = function(context, ...args) {
  // 如果 context 是 null/undefined，指向 window（非严格模式兼容）
  context = context || window;
  // 生成一个唯一的临时 key，防止覆盖对象原有属性
  const key = Symbol('fn');
  // 把当前函数（this）临时挂到 context 对象上
  context[key] = this;
  // 通过 context 调用 = this 指向 context
  const result = context[key](...args);
  // 执行完删除临时属性
  delete context[key];
  return result;
};

验证

二、手写 apply

源码（和 call 几乎一样，只是参数用数组接收）

// myApply(thisArg, [arg1, arg2, ...])
Function.prototype.myApply = function(context, argsArray) {
  context = context || window;
  const key = Symbol('fn');
  context[key] = this;
  // argsArray 可能是 undefined/null（不传参数的情况）
  const result = context[key](...(argsArray || []));
  delete context[key];
  return result;
};

验证

三、手写 bind

源码 — bind 比 call/apply 复杂，因为要处理预置参数 + new 调用

// myBind(thisArg, arg1, arg2, ...)
Function.prototype.myBind = function(context, ...presetArgs) {
  const originalFn = this; // 保存原始函数

  function boundFn(...callArgs) {
    // 判断是否通过 new 调用
    // new boundFn() → this instanceof boundFn = true
    const isNewCall = this instanceof boundFn;
    const finalCtx = isNewCall ? this : context;
    // 合并预置参数和实际调用时的参数
    return originalFn.apply(finalCtx, [...presetArgs, ...callArgs]);
  }

  // 维护原型链 — new 时能访问原函数的 prototype
  boundFn.prototype = Object.create(originalFn.prototype);
  return boundFn;
};

验证（含 new 调用场景）

四、手写 new

源码

// myNew(构造函数, ...参数)
function myNew(Constructor, ...args) {
  // 1. 创建一个新对象，原型指向 Constructor.prototype
  const obj = Object.create(Constructor.prototype);
  // 2. 执行构造函数，把 this 绑定到新对象
  const result = Constructor.apply(obj, args);
  // 3. 如果构造函数返回了对象（或函数），用它的返回值；否则用新对象
  return (result !== null && typeof result === 'object') ? result : obj;
}
