1. click / dblclick

click = 按下+松开一次；dblclick = 快速双击

el.addEventListener('click', (e) => {
  console.log('单击', e.clientX, e.clientY)
})
// 注意：dblclick 会先触发两次 click
// 触发顺序：mousedown → mouseup → click → dblclick

2. mousedown / mouseup

按下即触发 / 松开即触发，比 click 更底层

el.addEventListener('mousedown', () => {
  // 按住时触发，适合实现长按效果
})
// 通过 event.button 判断哪个键
// 0=左键 1=中键 2=右键

3. mousemove

鼠标在元素上移动时持续触发，频率很高，注意性能

el.addEventListener('mousemove', (e) => {
  console.log(`x:${e.clientX}, y:${e.clientY}`)
})
// 性能优化：用节流 throttle 或用 requestAnimationFrame
// mousemove 一秒钟可触发上百次

4. mouseenter/mouseleave vs mouseover/mouseout

关键区别：enter/leave 不冒泡，over/out 会冒泡

// ⭐ mouseenter / mouseleave：不冒泡
// 进入/离开元素本身，不受子元素影响
// mouseover / mouseout：会冒泡
// 鼠标移到子元素上也会触发父元素的 mouseout
// 大部分场景用 enter/leave 更符合预期

5. contextmenu — 右键菜单

右键触发，通常配合 preventDefault 屏蔽浏览器默认菜单

el.addEventListener('contextmenu', (e) => {
  e.preventDefault() // 阻止浏览器右键菜单
  // 这里显示自定义菜单...
})

6. wheel — 鼠标滚轮

deltaY > 0 向下滚，deltaY < 0 向上滚

el.addEventListener('wheel', (e) => {
  console.log('deltaY:', e.deltaY)
  // 正值=向下滚 负值=向上滚
}, { passive: false }) // 如需 e.preventDefault()

7. MouseEvent 常用属性

所有鼠标事件的 event 对象都包含这些属性

e.clientX / e.clientY
→ 相对于浏览器视口的坐标
e.pageX / e.pageY
→ 相对于文档的坐标（含滚动距离）
e.button
→ 0=左键 1=中键 2=右键
e.target / e.currentTarget
→ target=实际点击元素 / currentTarget=绑定事件的元素

二、键盘事件

⌨️ 点这里然后按键盘试试

试试字母 · 数字 · 方向键 · Ctrl+S · Shift+A · Escape · Enter

--:--:--等待按键...

1. keydown / keyup

keydown 按下触发（按住不放持续触发）；keyup 松开触发

document.addEventListener('keydown', (e) => {
  console.log('按下:', e.key)
})
// ⚠️ keypress 已废弃，不要使用
// keydown 在按住时会重复触发（repeat）

2. event.key — 按键名称（推荐）

返回按下键的字符串表示，大小写敏感

// 字母键
'a' 'A' // 大小写取决于是否按了 Shift
// 特殊键
'Enter' 'Escape' 'Tab' 'Backspace' 'Delete'
// 方向键
'ArrowUp' 'ArrowDown' 'ArrowLeft' 'ArrowRight'

3. event.code — 物理按键位置

code 代表键盘上的物理位置，与输入语言无关

// 无论输入法是什么，code 始终一致
'KeyA' 'Digit1' 'Space'
'Enter' 'Escape' 'Backspace'
'ArrowUp' 'ShiftLeft' 'ShiftRight'
// key vs code 区别：
// 按 Q 键：key='q', code='KeyQ'
// Shift+Q：key='Q', code='KeyQ'（code 不变！）

4. 修饰键：ctrlKey / shiftKey / altKey / metaKey

检查组合键，metaKey 在 Mac 是 Cmd，Win 是 Windows 键

// Ctrl+S 保存
document.addEventListener('keydown', (e) => {
  if (e.ctrlKey && e.key === 's') {
    e.preventDefault() // 阻止浏览器保存网页
    console.log('Ctrl+S 被拦截')
  }
})

5. 常用键盘实战场景

常见用例：快捷键、输入框控制、游戏方向

// 按 Escape 关闭弹窗
if (e.key === 'Escape') closeModal()
// 按 Enter 提交（输入框中）
if (e.key === 'Enter') form.submit()
// 阻止输入框输入数字以外的字符
if (!/[0-9]/.test(e.key) && e.key !== 'Backspace') e.preventDefault()

6. event.repeat — 是否重复触发

按住键不放时 keydown 会持续触发，repeat 为 true

document.addEventListener('keydown', (e) => {
  if (e.repeat) {
    console.log('按键持续按住中...')
  }
})

三、表单事件

📝 表单事件演示

--:--:--等待表单操作...

1. input — 实时输入

输入框内容变化时实时触发（包括粘贴、剪切、拖拽）

inputEl.addEventListener('input', (e) => {
  console.log('当前值:', e.target.value)
})
// 实战：搜索框实时搜索 / 字数统计 / 输入校验
// input 是最常用的表单事件

2. change — 值确认改变

用户完成修改并失去焦点时才触发（选择框则立即触发）

inputEl.addEventListener('change', (e) => {
  console.log('最终值:', e.target.value)
})
// ⚠️ input vs change 区别：
// input：每次输入都触发（实时）
// change：失去焦点且值变了才触发
// 对于 <select> / <input type="checkbox">：change 立即触发

3. focus / blur — 获得/失去焦点

focus 获得焦点触发，blur 失去焦点触发。不冒泡

inputEl.addEventListener('focus', () => {
  console.log('输入框获得焦点')
})
inputEl.addEventListener('blur', () => {
  console.log('输入框失去焦点，可以在这里做校验')
})
// focusin / focusout 是冒泡版本（用于事件委托）

4. submit — 表单提交

点击提交按钮或按 Enter 触发。通常用 preventDefault 阻止默认跳转

form.addEventListener('submit', (e) => {
  e.preventDefault() // 阻止页面刷新
  // 用 FormData 收集数据
  const data = new FormData(form)
  const obj = Object.fromEntries(data)
  console.log('提交数据:', obj)
})

5. reset 与事件触发顺序

reset 在表单重置时触发

form.addEventListener('reset', () => {
  console.log('表单已重置')
})
// 表单输入典型事件触发顺序：
// focus → input(多次) → change → blur

6. 事件委托（重要概念）

利用事件冒泡，在父元素上监听子元素的事件，减少监听器数量

// ❌ 低效：给每个按钮绑定
document.querySelectorAll('.btn').forEach(btn => {
  btn.addEventListener('click', handler)
})
// ✅ 高效：在父元素上做事件委托
parent.addEventListener('click', (e) => {
  if (e.target.matches('.btn')) handler(e)
})
→ 即使以后动态添加 .btn，也能响应
