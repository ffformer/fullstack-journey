01 — 作用域链（Scope Chain）

一、什么是作用域？

作用域决定了在代码的哪个位置可以访问某个变量。JS 有三种作用域：

  • 全局作用域：在任何函数外部声明的变量，所有代码都能访问
  • 函数作用域：在函数内部声明的变量，只有函数内部能访问
  • 块级作用域：用 let/const 声明的变量，只在 { } 块内有效

二、全局作用域 vs 函数作用域

示例：内外访问规则

var outer = '我在全局'; // 全局作用域 — 处处可访问

function showScope() {
  var inner = '我在函数内'; // 函数作用域 — 只有 showScope 内能访问
  console.log('函数内访问 outer：' + outer);   // ✅ 可以访问全局变量
  console.log('函数内访问 inner：' + inner);    // ✅ 可以访问自己的变量
}

showScope();
console.log('全局访问 inner：' + inner);  // ❌ ReferenceError: inner is not defined

三、块级作用域（let / const）

var 没有块级作用域，只有函数作用域。而 let 和 const 尊重 { } 块边界。

示例：var vs let 在块中的表现

if (true) {
  var a = 'var 声明';    // 无视块边界，在函数/全局生效
  let b = 'let 声明';    // 只在当前 {} 块内有效
}
console.log(a);  // ✅ 能访问到
console.log(b);  // ❌ ReferenceError: b is not defined

四、作用域链 — 查找规则

当代码访问一个变量时，JS 引擎按以下顺序查找：

  1. 当前块作用域
  2. 外层函数作用域（如果有）
  3. 再外层函数作用域...
  4. 全局作用域
  5. 找不到 → ReferenceError

这个由内到外的查找链路，就是作用域链。

示例：三层嵌套的作用域链

const global = '🌍 全局';

function outer() {
  const outerVar = '📦 outer';

  function inner() {
    const innerVar = '🎯 inner';
    // 从内到外逐层查找：自己的 innerVar → 外层的 outerVar → 全局的 global
    console.log(innerVar + ' → ' + outerVar + ' → ' + global);
  }

  inner();
}
outer();

五、容易踩的坑

坑1：循环中的 var

for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log('var i =', i), 100); // 输出 3, 3, 3 而非 0, 1, 2
}
// 原因：var 没有块作用域，所有回调共享同一个 i，循环结束时 i = 3

for (let j = 0; j < 3; j++) {
  setTimeout(() => console.log('let j =', j), 100); // 输出 0, 1, 2 ✅
}
// 原因：let 有块作用域，每次迭代都是一个新的 j 绑定

坑2：未声明直接赋值 = 隐式全局变量

function oops() {
  x = '我没有被声明！'; // 没有 var/let/const → JS 自动创建全局变量 x
}
oops();
console.log(x); // 😱 能访问到！这是 bug 的来源
